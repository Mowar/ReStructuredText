    :emphasize-lines: 3,5

        #!/bin/bash
        #================================================
        #FileName   :test_select_46_zhaojiedi.sh
        #Author     :zhaojiedi
        #Description:
        #DateTime   :2017-12-23 10:15:57
        #Version    :V1.0
        #Other      :
        #================================================

        select c in yes no ; do
                echo " you enter is $c"
                case $c in
                        yes)
                                echo "yes";;
                        no)
                                echo "no";;
                        *)
                                echo "other";;
                esac
        done
